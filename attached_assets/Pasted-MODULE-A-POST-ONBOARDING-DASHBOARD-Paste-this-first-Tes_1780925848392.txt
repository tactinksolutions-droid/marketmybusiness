javascriptimport express from 'express';
import { createClient } from '@supabase/supabase-js';
import dotenv from 'dotenv';
dotenv.config();

const router = express.Router();
const db = createClient(process.env.SUPABASE_URL, process.env.SUPABASE_SERVICE_KEY);

router.get('/me', async (req, res) => {
  res.json({ business: req.tenant });
});

// Channel connect — simulated for MVP, real OAuth in Phase 2
router.post('/connect/:platform', async (req, res) => {
  const { platform } = req.params;
  const validPlatforms = ['whatsapp', 'gmb', 'instagram', 'facebook'];
  if (!validPlatforms.includes(platform)) {
    return res.status(400).json({ error: 'Invalid platform' });
  }
  const field = `${platform}_connected`;
  const { data } = await db.from('businesses')
    .update({ [field]: true })
    .eq('id', req.tenant.id)
    .select().single();
  res.json({ success: true, platform, business: data });
});

// Channel disconnect
router.post('/disconnect/:platform', async (req, res) => {
  const { platform } = req.params;
  const validPlatforms = ['whatsapp', 'gmb', 'instagram', 'facebook'];
  if (!validPlatforms.includes(platform)) {
    return res.status(400).json({ error: 'Invalid platform' });
  }
  const field = `${platform}_connected`;
  await db.from('businesses').update({ [field]: false }).eq('id', req.tenant.id);
  res.json({ success: true, platform, connected: false });
});

// Full channel status
router.get('/channels', async (req, res) => {
  const b = req.tenant;
  res.json({
    channels: {
      whatsapp: { connected: b?.whatsapp_connected || false, label: 'WhatsApp Business' },
      instagram: { connected: b?.instagram_connected || false, label: 'Instagram' },
      gmb: { connected: b?.gmb_connected || false, label: 'Google My Business' },
      facebook: { connected: b?.facebook_connected || false, label: 'Facebook Pages' }
    }
  });
});

router.patch('/me', async (req, res) => {
  const allowed = ['language', 'brand_tone', 'brand_color', 'monthly_ad_budget', 'website_url'];
  const updates = Object.fromEntries(
    Object.entries(req.body).filter(([k]) => allowed.includes(k))
  );
  const { data } = await db.from('businesses').update(updates)
    .eq('id', req.tenant.id).select().single();
  res.json({ business: data });
});

export default router;

  async function handleFileUpload(e) {
    const file = e.target.files[0];
    if (!file) return;

    setImporting(true);
    setImportResult(null);

    // Parse CSV using simple split (no external library needed for basic CSVs)
    const text = await file.text();
    const lines = text.trim().split('\n');
    const headers = lines[0].toLowerCase().split(',').map(h => h.trim().replace(/"/g, ''));

    const nameIdx = headers.findIndex(h => h.includes('name'));
    const phoneIdx = headers.findIndex(h => h.includes('phone') || h.includes('mobile') || h.includes('number'));

    const contacts = lines.slice(1).map(line => {
      const cols = line.split(',').map(c => c.trim().replace(/"/g, ''));
      return {
        name: nameIdx >= 0 ? cols[nameIdx] : '',
        phone: phoneIdx >= 0 ? cols[phoneIdx] : '',
        email: emailIdx >= 0 ? cols[emailIdx] : '',
      };
    }).filter(c => c.phone || c.email);

    try {
      const { data } = await api.post('/contacts/import', { contacts });
      setImportResult({ success: true, count: data.imported });
      loadContacts();
    } catch (err) {
      setImportResult({ success: false, error: 'Import failed. Check your file format.' });
    } finally {
      setImporting(false);
      e.target.value = '';
    }
  }

  return (
    <div className="flex-1 overflow-y-auto bg-gray-50">
