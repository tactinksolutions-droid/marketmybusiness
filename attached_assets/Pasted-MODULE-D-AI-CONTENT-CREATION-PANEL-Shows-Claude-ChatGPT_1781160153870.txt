jsximport React, { useState } from 'react';
import api from '../lib/api';

const CONTENT_TYPES = [
  { id: 'whatsapp', label: 'WhatsApp message', icon: '💬', channel: 'whatsapp' },
  { id: 'instagram_caption', label: 'Instagram caption', icon: '📸', channel: 'instagram' },
  { id: 'linkedin_post', label: 'LinkedIn post', icon: '💼', channel: 'linkedin' },
  { id: 'email_subject', label: 'Email subject + body', icon: '📧', channel: 'email' },
  { id: 'gmb_post', label: 'Google My Business post', icon: '📍', channel: 'gmb' },
  { id: 'facebook_post', label: 'Facebook post', icon: '👥', channel: 'facebook' },
  { id: 'google_ad', label: 'Google Ad copy', icon: '🎯', channel: 'google_ads' },
];

const AI_ENGINES = [
  {
    id: 'claude',
    name: 'Claude',
    company: 'Anthropic',
    icon: '🧠',
    strength: 'Brand voice, strategy, nuanced writing',
    color: 'violet',
    active: true,
  },
  {
    id: 'chatgpt',
    name: 'ChatGPT',
    company: 'OpenAI',
    icon: '💡',
    strength: 'Creative ideation, SEO copy, variations',
    color: 'emerald',
    active: false,
  },
  {
    id: 'gemini',
    name: 'Gemini',
    company: 'Google',
    icon: '✨',
    strength: 'Multilingual, image prompts, video scripts',
    color: 'blue',
    active: false,
  },
];

export default function ContentPage({ business }) {
  const [brief, setBrief] = useState('');
  const [contentType, setContentType] = useState('whatsapp');
  const [results, setResults] = useState({});
  const [loading, setLoading] = useState({});
  const [selectedEngines, setSelectedEngines] = useState(['claude']);

  function toggleEngine(id) {
    setSelectedEngines(prev =>
      prev.includes(id) ? prev.filter(e => e !== id) : [...prev, id]
    );
  }

  async function generateContent() {
    if (!brief.trim()) return;

    const type = CONTENT_TYPES.find(t => t.id === contentType);
    setResults({});

    for (const engineId of selectedEngines) {
      setLoading(prev => ({ ...prev, [engineId]: true }));
      try {
        const { data } = await api.post('/content/generate', {
          engine: engineId,
          brief,
          contentType,
          channel: type.channel,
        });
        setResults(prev => ({ ...prev, [engineId]: data.content }));
      } catch (err) {
        setResults(prev => ({
          ...prev,
          [engineId]: `Unable to generate — ${engineId === 'claude' ? 'check API key' : 'API key required in Integrations'}`
        }));
      } finally {
        setLoading(prev => ({ ...prev, [engineId]: false }));
      }
    }
  }

  async function useDraft(content, channel) {
    await api.post('/chat', {
      message: `Use this as the campaign message for ${channel}: "${content}"`
    });
    alert(`Draft sent to chat. Go to AI Chat to review and send.`);
  }

  const activeEngines = AI_ENGINES.filter(e => selectedEngines.includes(e.id));

  return (
    <div className="flex-1 overflow-y-auto bg-gray-50">
      <div className="max-w-4xl mx-auto px-6 py-8">

        <div className="mb-6">
          <h1 className="text-xl font-semibold text-gray-900 mb-1">AI Content Studio</h1>
          <p className="text-sm text-gray-500">
            Generate marketing content for any channel using multiple AI engines simultaneously.
          </p>
        </div>

        <div className="bg-white rounded-2xl border border-gray-200 p-5 mb-5">

          {/* Content type selector */}
          <div className="mb-4">
            <label className="block text-xs font-medium text-gray-600 mb-2 uppercase tracking-wide">Content type</label>
            <div className="flex flex-wrap gap-2">
              {CONTENT_TYPES.map(type => (
                <button key={type.id} onClick={() => setContentType(type.id)}
                  className={`flex items-center gap-1.5 text-xs font-medium px-3 py-1.5 rounded-full border transition-colors
                    ${contentType === type.id
                      ? 'bg-emerald-700 text-white border-emerald-700'
                      : 'bg-white text-gray-600 border-gray-200 hover:bg-gray-50'}`}>
                  <span>{type.icon}</span> {type.label}
                </button>
              ))}
            </div>
          </div>

          {/* Brief input */}
          <div className="mb-4">
            <label className="block text-xs font-medium text-gray-600 mb-2 uppercase tracking-wide">Campaign brief</label>
            <textarea
              value={brief}
              onChange={e => setBrief(e.target.value)}
              rows={3}
              placeholder={`Describe what you want to communicate. Example: "Promote our Zen Garden home installation for the upcoming new year. Calm, inspiring tone. Offer a free home visit consultation."`}
              className="w-full border border-gray-200 rounded-xl px-4 py-3 text-sm focus:outline-none focus:ring-2 focus:ring-emerald-500 resize-none"
            />
          </div>

          {/* AI Engine selector */}
          <div className="mb-5">
            <label className="block text-xs font-medium text-gray-600 mb-2 uppercase tracking-wide">Generate with</label>
            <div className="flex gap-2 flex-wrap">
              {AI_ENGINES.map(engine => (
                <button key={engine.id} onClick={() => toggleEngine(engine.id)}
                  className={`flex items-center gap-2 px-3 py-2 rounded-xl border text-sm transition-all
                    ${selectedEngines.includes(engine.id)
                      ? 'bg-gray-900 text-white border-gray-900'
                      : 'bg-white text-gray-600 border-gray-200 hover:bg-gray-50'}`}>
                  <span>{engine.icon}</span>
                  <span className="font-medium">{engine.name}</span>
                  {!engine.active && !selectedEngines.includes(engine.id) && (
                    <span className="text-xs text-gray-400">Key needed</span>
                  )}
                </button>
              ))}
            </div>
          </div>

          <button onClick={generateContent}
            disabled={!brief.trim() || selectedEngines.length === 0}
            className="w-full bg-emerald-700 text-white py-3 rounded-xl text-sm font-semibold hover:bg-emerald-800 disabled:opacity-40 transition-colors">
            Generate content for {selectedEngines.length} AI engine{selectedEngines.length !== 1 ? 's' : ''}
          </button>
        </div>

        {/* Results — side by side */}
        {(Object.keys(results).length > 0 || Object.values(loading).some(Boolean)) && (
          <div className={`grid gap-4 ${activeEngines.length > 1 ? 'grid-cols-2' : 'grid-cols-1'}`}>
            {activeEngines.map(engine => (
              <div key={engine.id} className="bg-white rounded-2xl border border-gray-200 overflow-hidden">

                <div className="px-4 py-3 border-b border-gray-100 flex items-center gap-2">
                  <span className="text-lg">{engine.icon}</span>
                  <div>
                    <p className="text-sm font-semibold text-gray-900">{engine.name}</p>
                    <p className="text-xs text-gray-400">{engine.strength}</p>
                  </div>
                </div>

                <div className="p-4">
                  {loading[engine.id] ? (
                    <div className="flex items-center gap-2 text-sm text-gray-400 py-4">
                      <div className="w-4 h-4 border-2 border-emerald-500 border-t-transparent rounded-full animate-spin"></div>
                      Generating...
                    </div>
                  ) : results[engine.id] ? (
                    <>
                      <p className="text-sm text-gray-700 leading-relaxed whitespace-pre-wrap mb-4">
                        {results[engine.id]}
                      </p>
                      <div className="flex gap-2">
                        <button
                          onClick={() => navigator.clipboard.writeText(results[engine.id])}
                          className="flex-1 border border-gray-200 text-gray-600 text-xs font-medium py-2 rounded-lg hover:bg-gray-50 transition-colors">
                          Copy
                        </button>
                        <button
                          onClick={() => useDraft(results[engine.id], CONTENT_TYPES.find(t => t.id === contentType)?.channel)}
                          className="flex-1 bg-emerald-700 text-white text-xs font-medium py-2 rounded-lg hover:bg-emerald-800 transition-colors">
                          Use this draft
                        </button>
                      </div>
                    </>
                  ) : (
                    <p className="text-sm text-gray-400 py-4">Waiting...</p>
                  )}
                </div>
              </div>
            ))}
          </div>
        )}
      </div>
    </div>
  );
}

=== ADD Content generation route: server/routes/content.js ===

javascriptimport express from 'express';
import Anthropic from '@anthropic-ai/sdk';
import axios from 'axios';
import dotenv from 'dotenv';
dotenv.config();

const router = express.Router();
const anthropic = new Anthropic({ apiKey: process.env.ANTHROPIC_API_KEY });

router.post('/generate', async (req, res) => {
  const { engine, brief, contentType, channel } = req.body;
  const business = req.tenant;

  const systemPrompt = `You are a marketing copywriter for ${business.name}, a ${business.type} in ${business.city}, India.
Brand tone: ${business.brand_tone || 'professional, warm'}.
Website: ${business.website_url || ''}.
Write ONLY the content requested — no preamble, no explanation, no labels.`;

  const userPrompt = `Write a ${contentType.replace('_', ' ')} for ${channel}.
Brief: ${brief}
Write in the brand's tone. For Indian audience. Keep it authentic and non-generic.`;

  try {
    if (engine === 'claude') {
      const response = await anthropic.messages.create({
        model: 'claude-sonnet-4-20250514',
        max_tokens: 600,
        system: systemPrompt,
        messages: [{ role: 'user', content: userPrompt }],
      });
      return res.json({ content: response.content[0].text, engine: 'claude' });
    }

    if (engine === 'chatgpt') {
      const { data: biz } = await require('../lib/supabaseAdmin.js').supabaseAdmin
        .from('businesses').select('openai_api_key').eq('id', business.id).single();

      const key = biz?.openai_api_key || process.env.OPENAI_API_KEY;
      if (!key) return res.status(400).json({ error: 'OpenAI API key not configured' });

      const response = await axios.post('https://api.openai.com/v1/chat/completions', {
        model: 'gpt-4o-mini',
        messages: [
          { role: 'system', content: systemPrompt },
          { role: 'user', content: userPrompt },
        ],
        max_tokens: 600,
      }, {
        headers: { Authorization: `Bearer ${key}`, 'Content-Type': 'application/json' },
      });
      return res.json({ content: response.data.choices[0].message.content, engine: 'chatgpt' });
    }

    if (engine === 'gemini') {
      const { data: biz } = await require('../lib/supabaseAdmin.js').supabaseAdmin
        .from('businesses').select('gemini_api_key').eq('id', business.id).single();

      const key = biz?.gemini_api_key || process.env.GEMINI_API_KEY;
      if (!key) return res.status(400).json({ error: 'Gemini API key not configured' });

      const response = await axios.post(
        `https://generativelanguage.googleapis.com/v1beta/models/gemini-pro:generateContent?key=${key}`,
        {
          contents: [{
            parts: [{ text: `${systemPrompt}\n\n${userPrompt}` }]
          }]
        }
      );
      const text = response.data.candidates[0].content.parts[0].text;
      return res.json({ content: text, engine: 'gemini' });
    }

    res.status(400).json({ error: 'Unknown engine' });

  } catch (err) {
    console.error(`${engine} generation error:`, err.message);
    res.status(500).json({ error: err.message });
  }
});

export default router;
