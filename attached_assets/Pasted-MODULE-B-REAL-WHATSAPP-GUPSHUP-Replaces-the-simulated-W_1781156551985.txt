javascriptimport axios from 'axios';
import { createClient } from '@supabase/supabase-js';
import dotenv from 'dotenv';
dotenv.config();

const db = createClient(process.env.SUPABASE_URL, process.env.SUPABASE_SERVICE_KEY);

// Get Gupshup API key for this business (stored encrypted in DB)
async function getGupshupKey(businessId) {
  const { data } = await db.from('businesses')
    .select('gupshup_api_key').eq('id', businessId).single();
  return data?.gupshup_api_key || process.env.GUPSHUP_API_KEY;
}

// Send a single WhatsApp message via Gupshup
export async function sendWhatsAppMessage(businessId, phone, message, sourcePhone) {
  const apiKey = await getGupshupKey(businessId);
  if (!apiKey) throw new Error('WhatsApp not configured');

  // Normalise Indian phone number
  const normalised = phone.replace(/\D/g, '');
  const e164 = normalised.startsWith('91') ? normalised : `91${normalised}`;

  const response = await axios.post(
    'https://api.gupshup.io/sm/api/v1/msg',
    new URLSearchParams({
      channel: 'whatsapp',
      source: sourcePhone || process.env.GUPSHUP_SOURCE_NUMBER,
      destination: e164,
      message: JSON.stringify({ type: 'text', text: message }),
      'src.name': process.env.GUPSHUP_APP_NAME || 'MarketMyBusiness',
    }),
    {
      headers: {
        apikey: apiKey,
        'Content-Type': 'application/x-www-form-urlencoded',
      },
    }
  );

  return response.data;
}

// Send a campaign to a list of contacts
export async function sendCampaign(businessId, campaignId, contacts, messageBody) {
  const results = { sent: 0, failed: 0, errors: [] };

  for (const contact of contacts) {
    if (!contact.phone) continue;
    try {
      // Personalise message — replace [Name] with contact name
      const personalised = messageBody
        .replace(/\[Name\]/gi, contact.name?.split(' ')[0] || 'there')
        .replace(/\[name\]/gi, contact.name?.split(' ')[0] || 'there');

      await sendWhatsAppMessage(businessId, contact.phone, personalised);
      results.sent++;

      // Small delay between messages to avoid rate limiting
      await new Promise(r => setTimeout(r, 200));
    } catch (err) {
      results.failed++;
      results.errors.push({ phone: contact.phone, error: err.message });
    }
  }

  // Update campaign stats
  await db.from('campaigns').update({
    status: 'sent',
    sent_at: new Date().toISOString(),
    total_sent: results.sent,
    total_delivered: results.sent, // Updated via webhook later
  }).eq('id', campaignId);

  return results;
}

javascriptimport express from 'express';
import { createClient } from '@supabase/supabase-js';
import { sendCampaign } from '../services/whatsappService.js';
import dotenv from 'dotenv';
dotenv.config();

const router = express.Router();
const db = createClient(process.env.SUPABASE_URL, process.env.SUPABASE_SERVICE_KEY);

// Save API credentials for a channel
router.post('/connect', async (req, res) => {
  const { channel, credential } = req.body;
  const bizId = req.tenant.id;

  const fieldMap = {
    whatsapp: { key: 'gupshup_api_key', connected: 'whatsapp_connected' },
    email: { key: 'brevo_api_key', connected: 'email_connected' },
    chatgpt: { key: 'openai_api_key', connected: 'chatgpt_connected' },
    gemini: { key: 'gemini_api_key', connected: 'gemini_connected' },
  };

  const fields = fieldMap[channel];
  if (!fields) return res.status(400).json({ error: 'Unknown channel' });

  await db.from('businesses').update({
    [fields.key]: credential,
    [fields.connected]: true,
  }).eq('id', bizId);

  res.json({ success: true, channel, connected: true });
});

// Execute an approved campaign — actually sends messages
router.post('/campaigns/:id/send', async (req, res) => {
  const bizId = req.tenant.id;

  const { data: campaign } = await db.from('campaigns')
    .select('*').eq('id', req.params.id).eq('business_id', bizId).single();

  if (!campaign) return res.status(404).json({ error: 'Campaign not found' });

  // Get contacts for this segment
  let query = db.from('contacts').select('*').eq('business_id', bizId);
  if (campaign.segment !== 'all') {
    query = query.eq('segment', campaign.segment);
  }
  const { data: contacts } = await query;

  if (!contacts?.length) {
    return res.status(400).json({ error: 'No contacts in this segment' });
  }

  // Route to correct channel
  if (campaign.channel === 'whatsapp') {
    const biz = req.tenant;
    if (!biz.whatsapp_connected || !biz.gupshup_api_key) {
      // Simulated send for demo — WhatsApp not yet configured
      await db.from('campaigns').update({
        status: 'sent',
        sent_at: new Date().toISOString(),
        total_sent: contacts.length,
        total_delivered: contacts.length,
        total_read: Math.floor(contacts.length * 0.68),
      }).eq('id', campaign.id);

      return res.json({
        success: true,
        mode: 'simulated',
        message: `Campaign sent to ${contacts.length} contacts (simulated — connect WhatsApp API key to send real messages)`,
        stats: { sent: contacts.length, delivered: contacts.length }
      });
    }

    // REAL SEND via Gupshup
    const results = await sendCampaign(bizId, campaign.id, contacts, campaign.message_body);
    return res.json({ success: true, mode: 'live', ...results });
  }

  if (campaign.channel === 'email') {
    const results = await sendEmailCampaign(req.tenant, campaign, contacts);
    return res.json({ success: true, mode: 'live', ...results });
  }

  // Instagram, LinkedIn, Facebook — simulated for now
  await db.from('campaigns').update({
    status: 'sent',
    sent_at: new Date().toISOString(),
    total_sent: 1,
  }).eq('id', campaign.id);

  res.json({
    success: true,
    mode: 'simulated',
    message: `Post scheduled for ${campaign.channel}. Real publishing coming soon.`,
  });
});

// Gupshup delivery webhook (public — no tenant auth)
router.post('/webhook/whatsapp', async (req, res) => {
  await handleDeliveryWebhook(req.body);
  res.json({ status: 'ok' });
});

export default router;
